    /**
     * Count the number of occurrences of each word contained in docWords.
     *
     * @param docWords container of words
     *
     * @return a correspondence of words and their number of occurrences in docWords
     */
    public Map<String, Integer> countWords(final AbstractList<String> docWords)
    {
        final Map<String, Integer> wordCounts = new Hashtable<>();

        docWords.forEach((word) -> {
            if (wordCounts.get(word) == null)
            {
                // first occurrence encountered of word
                wordCounts.put(word, 1);
            }
            else
            {
                // word has already been encountered
                // increment its recorded occurrences
                final Integer currentWordCount = wordCounts.get(word);
                wordCounts.replace(word, currentWordCount + 1);
            }
        });

        return wordCounts;
    }
