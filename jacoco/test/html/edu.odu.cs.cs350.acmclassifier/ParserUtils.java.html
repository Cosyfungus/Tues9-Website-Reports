package edu.odu.cs.cs350.acmclassifier;

import java.io.FileNotFoundException;
import java.io.IOException;

import java.util.AbstractList;

import org.apache.tika.exception.TikaException;

import org.xml.sax.SAXException;

final public class ParserUtils
{
		public static AbstractList<String> getStopWords(final String stopWordsFilePath) throws FileNotFoundException, IOException, SAXException, TikaException, IllegalArgumentException
		{
			return DocumentParserUtils.parse(stopWordsFilePath).getWords();
		}

		// removes stop words using the "removeAll" function of ArrayLists
		public static AbstractList<String> removeStopWords(final AbstractList<String> stopWords, final AbstractList<String> docWords)
		{
            final AbstractList<String> docWordsAltered = docWords;

		    docWordsAltered.removeAll(stopWords);

		    return docWordsAltered;
		}

        // utility classes require constructor be declared private to prevent instantiation
        private ParserUtils() {}
}
