package edu.odu.cs.cs350.acmclassifier;

import java.lang.String;
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.File;
import java.io.FileNotFoundException;
import org.apache.tika.exception.TikaException;
import org.xml.sax.SAXException;

/*
	 * Driver to Improve Training Data 
	 * 
	 * @author Justin Jenkins
	 */
public class TrainerDriver {
    
    	/*
	 * Retrieves the file path for the document(s) to be classified
     * as well as training data for the ML section
     * 
     * For now the same as ClassifyDriver and will be modified when I know how
     * the training data will come in
     * 
	 * Runs necessary functions from other classes
	 * 
	 * @param args should be paths to each file to be classified
	 *
	 * @author Justin Jenkins
	 */
    public static void main(String[] args) throws FileNotFoundException, IOException, SAXException, TikaException, IllegalArgumentException
	{
		Document doc;
		//if no args are passed it will accept a path from console input
		if (args.length == 0)
		{
			String path = getPath();
			String category = getCategory(); //for future use on trainer

			if(validatePath(path)) {doc = DocumentParserUtils.parse(path);}
			
		}
		//will process each argument given
		else
		{
			for (String path : args)
			{
				if(validatePath(path)) {doc = DocumentParserUtils.parse(path);}
			}
		}


		System.out.println("Placeholder");
	}
    /* Function to grab a file path from console-based input
    * 
    * @return returns path given from console input
    * 
    * @author Justin Jenkins
    */
    public static String getPath() throws IOException
	{
		System.out.println("Enter a Path: ");
		BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
		String path = reader.readLine();
		return path;
	}

	/*Function to grab the category for training crom console-based input
	 * 
	 * @return returns path given from console input
	 * 
	 * @author Justin Jenkins
	 */
	public static String getCategory() throws IOException
	{
		System.out.println("Enter the category: ");
		BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
		String category = reader.readLine();
		return category;
	}

	/*
	 * Validates that the path leads to a valid file that can be read
	 * 
	 * @param path to the file
	 * @author Justin Jenkins
	 */
	private static boolean validatePath(String path)
	{
		try {
			File testFile = new File(path);
			if (testFile.canRead() == false)
			{
				System.out.println("File cannot be read.");
				return false;
			}
		
		}catch (NullPointerException e)
			{ 
				System.out.println("Path was null." + e.toString());
				return false;
			}
		return true;
	}
}
