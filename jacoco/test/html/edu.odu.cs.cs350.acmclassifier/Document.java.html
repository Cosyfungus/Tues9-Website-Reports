package edu.odu.cs.cs350.acmclassifier;

import java.util.AbstractList;
import java.util.ArrayList;

/**
 * Object to store document contents.
 * 
 * @author tr29038
 *
 */
public class Document 
{
    // Words contained in document file
	private final AbstractList<String> words;

	public Document(final AbstractList<String> docWords)
	{
		words = docWords;
	}
	
	/**
	 * @return container of words contained
	 */
	public AbstractList<String> getWords()
	{
		return new ArrayList<>(words);
	}
	
	/**
	 * @return number of words contained
	 */
	public long getWordCount()
	{
		return words.size();
	}
	
	// Prevent default Document construction -- should never be executed
	private Document() { words = new ArrayList<>(); }
}
